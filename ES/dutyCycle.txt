ORG 0000H        ; Set program start address
MAIN: 
    SETB P1.0     ; Set P1.0 High
    ACALL DELAY   ; Call delay
    CLR P1.0      ; Set P1.0 Low
    ACALL DELAY   ; Call delay
    SJMP MAIN     ; Repeat forever

; --- Delay Subroutine (approx. 500ms for 12MHz)
DELAY:
    MOV R2, #20
LOOP1: MOV R1, #255
LOOP2: MOV R0, #255
LOOP3: DJNZ R0, LOOP3
       DJNZ R1, LOOP2
       DJNZ R2, LOOP1
       RET

END              ; End of program
